A:
	s2 = s1 - ((n*8 + 30) & 0xfffffffffffffff0)
	即 s2 = s1 - ((n*8+30)>>4<<4)
	当n是偶数时：
		(8n + 30)>>4<<4 = (n/2 + 1)<<4 = 8n + 16
		s2 = s1 - (8n + 16)
	当n时奇数时：
		(8n + 30)>>4<<4 = (8(n-1) + 38)>>4<<4 = ((n-1)/2 + 2)<<4 = 8(n-1) + 32 = 8n + 24
		s2 = s1 - (8n + 24)

B:
	(15 + %rsp)>>4<<4 = %rsp>>4<<4 或 (%rsp>>4 + 1)<<4,
	使%r8中的值满足大于等于%rsp，并且是最接近于%rsp的16的倍数，使数组的起始地址是16的倍数

	e1最大为24：n是奇数，s1 = 16x	   其中x是整数且16x > 8n + 24
